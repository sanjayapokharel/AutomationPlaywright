pipeline {
    agent any

    environment {
        NODE_ENV = 'test'
    }

    stages {

        stage('Checkout Source Code') {
            steps {
                echo 'Checking out source code from GitHub...'
                checkout scm
            }
        }

        stage('Verify Node & NPM') {
            steps {
                echo 'Verifying Node.js and npm versions...'
                bat 'node -v'
                bat 'npm -v'
            }
        }

        stage('Install Project Dependencies') {
            steps {
                echo 'Installing npm dependencies...'
                bat 'npm install'
            }
        }

        stage('Install Playwright Browsers') {
            steps {
                echo 'Installing Playwright browsers...'
                bat 'npx playwright install'
            }
        }

        stage('Run Playwright Tests') {
            steps {
                echo 'Executing Playwright tests...'
                bat 'npx playwright test'
            }
        }

        stage('Generate Playwright Report') {
            steps {
                echo 'Playwright HTML report generated automatically'
            }
        }

        stage('Deploy') {
            steps {
                echo 'Deploy stage (placeholder)'
            }
        }
    }

    post {

        always {
            echo 'Archiving Playwright HTML report...'
            archiveArtifacts artifacts: 'playwright-report/**', fingerprint: true
         emailext(
            subject: "Playwright Test Report - ${currentBuild.currentResult}",
            mimeType: 'text/html',
            to: 'yourmail@gmail.com',
            body: """
                <h3>Playwright Automation Report</h3>
                <p>Status: <b>${currentBuild.currentResult}</b></p>
                <p>Job: ${env.JOB_NAME}</p>
                <p>Build Number: ${env.BUILD_NUMBER}</p>
                <p>
                  <a href="${env.BUILD_URL}">
                    View Jenkins Build
                  </a>
                </p>
            """,
            attachmentsPattern: 'playwright-report/**'
        )
        }

        success {
            echo '✅ Pipeline executed successfully'
        }

        failure {
            echo '❌ Pipeline failed'
        }
    }
}
