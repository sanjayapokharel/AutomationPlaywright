pipeline {
    agent any

    environment {
        NODE_ENV = 'test'
    }

    stages {

        // üîî Notify Build Started
        stage('Notify Build Started') {
            steps {
                echo 'Sending Build Started email...'
                emailext(
                    subject: "üöÄ Build Started: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                    body: """
Hello Team,

A Jenkins pipeline has started.

Job Name     : ${env.JOB_NAME}
Build Number : #${env.BUILD_NUMBER}
Node         : ${env.NODE_NAME}
Environment  : Localhost Jenkins

Build URL:
${env.BUILD_URL}

Regards,
Jenkins CI
""",
                    to: "psanzay11@gmail.com, dpesadhikari@gmail.com"
                )
            }
        }

        // üîπ Checkout Code
        stage('Checkout Source Code') {
            steps {
                echo 'Checking out source code from Git...'
                checkout scm
            }
        }

        // üîπ Verify Node & NPM
        stage('Verify Node & NPM') {
            steps {
                echo 'Checking Node.js and npm versions...'
                bat 'node -v'
                bat 'npm -v'
            }
        }

        // üîπ Install Dependencies
        stage('Install Project Dependencies') {
            steps {
                echo 'Installing npm dependencies...'
                bat 'npm ci'
            }
        }

        // üîπ Install Playwright Browsers
        stage('Install Playwright Browsers') {
            steps {
                echo 'Installing Playwright browsers...'
                bat 'npx playwright install'
            }
        }

        // üîπ Run Playwright Tests & Generate HTML Report
        stage('Run Playwright Tests') {
            steps {
                echo 'Executing Playwright tests...'
                // Ensure Playwright HTML report is generated
                bat 'npx playwright test --reporter=html'
            }
        }
    }

    post {
        always {
            echo 'Archiving Playwright HTML report...'

            // Archive report artifacts
            archiveArtifacts artifacts: 'playwright-report/**', fingerprint: true

            // Publish HTML report in Jenkins UI
            publishHTML([
                reportDir: 'playwright-report',
                reportFiles: 'index.html',
                reportName: 'Playwright Report',
                keepAll: true,
                alwaysLinkToLastBuild: true
            ])
        }

        success {
            emailext(
                subject: "‚úÖ Build Success: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                body: """
Good news üéâ

The Jenkins pipeline completed successfully.

Job Name     : ${env.JOB_NAME}
Build Number : #${env.BUILD_NUMBER}

Playwright Report:
${env.BUILD_URL}Playwright_Report/

Build URL:
${env.BUILD_URL}

Regards,
Jenkins CI
""",
                to: "psanzay11@gmail.com, dpesadhikari@gmail.com"
            )
        }

        failure {
            emailext(
                subject: "‚ùå Build Failed: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                body: """
Attention ‚ùó

The Jenkins pipeline has FAILED.

Job Name     : ${env.JOB_NAME}
Build Number : #${env.BUILD_NUMBER}

Check logs and Playwright report:
${env.BUILD_URL}Playwright_Report/

Regards,
Jenkins CI
""",
                to: "psanzay11@gmail.com, dpesadhikari@gmail.com"
            )
        }
    }
}
